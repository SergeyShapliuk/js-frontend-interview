function binarySearch(arr, target) {
  let left = 0;
  let right = arr.length - 1;
  
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    
    if (arr[mid] === target) {
      return mid;  // Ğ½Ğ°Ğ¹Ğ´ĞµĞ½Ğ¾
    } else if (arr[mid] < target) {
      left = mid + 1;  // Ğ¸Ñ‰ĞµĞ¼ ÑĞ¿Ñ€Ğ°Ğ²Ğ°
    } else {
      right = mid - 1;  // Ğ¸Ñ‰ĞµĞ¼ ÑĞ»ĞµĞ²Ğ°
    }
  }
  
  return -1;  // Ğ½Ğµ Ğ½Ğ°Ğ¹Ğ´ĞµĞ½Ğ¾

function bubbleSort(arr) {
  const n = arr.length;
  
  for (let i = 0; i < n - 1; i++) {
    for (let j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        // ĞœĞµĞ½ÑĞµĞ¼ Ğ¼ĞµÑÑ‚Ğ°Ğ¼Ğ¸
        let temp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = temp;
      }
    }
  }
  
  return arr;
}

// ĞŸÑ€Ğ¸Ğ¼ĞµÑ€:

  let max = arr[0];
  
  for (let i = 1; i < arr.length; i++) {
    if (arr[i] > max) {
      max = arr[i];
    }
  }
  
  return max;
}
